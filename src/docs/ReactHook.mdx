## React Hook

React users can use the hook useAnimare by importing it from animare/react. This hook returns the animation's object to use it in your React component.

```js title="App.jsx"
import animare, { ease } from 'animare';
import { useAnimare } from 'animare/react';

function MyComponent() {
  const ballAnimation = useAnimare(() => {
    const ball = document.getElementById('ball');

    const options = {
      to: 1,
      duration: 1000,
      ease: ease.out.quad,
      autoPlay: false,
    };

    const onUpdate = ([scale]) => {
      ball.style.transform = `scale(${scale})`;
    };

    return animare(options, onUpdate);
  });

  const onClick = () => {
    ballAnimation?.play();
  };

  // ...
}
```
