import { Tabs, TabItem } from '@astrojs/starlight/components';

Install using your package manager of choice to add animare to your project:

<Tabs>
    <TabItem label='npm'>
      ```bash frame="none"
      npm install animare 
      ```
    </TabItem>

    <TabItem label='pnpm'>
      ```bash frame="none"
      pnpm add animare
      ```
    </TabItem>

    <TabItem label='yarn'>
      ```bash frame="none"
      yarn add animare
      ```
    </TabItem>

</Tabs>

<br/>

Aaaaand, you’re done! That was fast. 🐇
